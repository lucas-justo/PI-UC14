CREATE TABLE produto
(
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    nome VARCHAR(50) , 
    preco DOUBLE ,
    quantidade INT ,
    idCategoria INT ,
    foto VARCHAR(100) ,
    descricao VARCHAR(400)  ,
    marca VARCHAR(50)   NULL ,
    promo INT NULL
);

CREATE TABLE cidade (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    nome VARCHAR(50) ,
    idEstado INT  
);

CREATE TABLE estado (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    nome VARCHAR(50) ,
    idPais INT
);

CREATE TABLE pais (
    id INT NOT NULL PRIMARY KEY AUTO_INCREMENT ,
    nome VARCHAR(50) ,
    lingua VARCHAR(30),
    moeda VARCHAR(30)
);
